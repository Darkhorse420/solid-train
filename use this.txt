#include <msp430.h> 
#include <stdio.h>
#include <stdint.h>
#define GroupNum 0x48

extern void init_mydata(void);
extern uint8_t myData(uint8_t);



int main(void)
{
    WDTCTL = WDTPW | WDTHOLD;   // stop timer

    volatile uint16_t vmax, vmin, vmed, vCount, vSum, vavg, result, temp;
    volatile int swap, k;
    int sortie[256];

    init_myData();
    vCount = 0;
    vSum = 0;

    result=myData(GroupNum);
        while(result!=0x00)
        {
        vSum += result;
        sortie[vCount]=result;
        vCount++;
        result = myData(GroupNum);
        }

        do
        {   swap=0;
            for(k=0;k<vCount;k++)
            {
                if(sortie[k]>sortie[k+1])
                {
                    temp=sortie[k];
                    sortie[k]=sortie[k+1];
                    sortie[k+1]=temp;
                    swap=1;
                }
            }


        }while(swap);


    vmax=sortie[vCount];
    vmin=sortie[0];
    vmed=sortie[vCount/2];
    vavg=vSum/vCount;


    printf("Dominic Harshaw\nFall 2017\nM4-- Problem #A\n");
    printf("------------------------\n");
    printf("vMax = %d\n", vmax);
    printf("vMin = %d\n", vmin);
    printf("vMed = %d\n", vmed);
    printf("vAvg = %d\n", vavg);



    return 0;
}
